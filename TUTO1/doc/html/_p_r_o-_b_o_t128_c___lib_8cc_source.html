560 //---[ GO_TURN(Distance "cm", Degree "Grad", Speed 1 To 255) ]------------------
561 void GO_TURN(int distance, int degree, byte speed)
562 {
563  int encoder_cnt, drive_cnt, diff, speed_left, speed_right;
564 
565  //Distance: postiv=go forward / negativ=go backward / zero=use degree for turn
566  //Degree : postiv=turn right / negativ=turn left
567  //Speed : sets motorspeed
568 
569  drive_cnt=0;
570  encoder_cnt=0;
571  diff=0;
572  speed_left=speed;
573  speed_right=speed;
574 
575  //Stop the robot
576  MOTOR_STOP();
577 
578  //if distance is Not zero, Then take this value To go ...
579  if (distance!=0)
580  {
581  //Calculate ticks "cm"
582  encoder_cnt=ABS_INT((distance-3))*UNIT_TICK;
583  if (distance<0)
584  {
585  //Backward
586  MOTOR_DIR(0,0);
587  } //Forward
588  else MOTOR_DIR(1,1);
589  }
590  else
591  {
592  //calculate degree
593  encoder_cnt=ABS_INT(degree)*10;
594  encoder_cnt=(encoder_cnt/37)-3;
595  if (degree<0)
596  {
597  //Turn left
598  MOTOR_DIR(0,1);
599  } //Turn right
600  else MOTOR_DIR(1,0);
601  }
602 
603  //Reset encoder
604  ODO_RESET();
605 
606  //Now move the bot
607  MOTOR_POWER(speed_left,speed_right);
608 
609  do
610  {
611  drive_cnt=drive_cnt+((ODO_LEFT+ODO_RIGHT)/2);
612  diff=ENCODER_LEFT-ENCODER_RIGHT;
613 
614  if (diff>0)
615  {
616  //Left faster than right
617  if ((speed_left>speed) | (speed_right>244))
618  {
619  speed_left=speed_left-SPEED_INTEGRATOR;
620  }
621  else speed_right=speed_right+SPEED_INTEGRATOR;
622  }
623 
624  if (diff<0)
625  {
626  //Right faster than left
627  if ((speed_right>speed) | (speed_left>244))
628  {
629  speed_right=speed_right-SPEED_INTEGRATOR;
630  }
631  else speed_left=speed_left+SPEED_INTEGRATOR;
632  }
633 
634  //Reset encoder
635  ODO_RESET();
636 
637  //Set Motorpower
638  MOTOR_POWER(speed_left,speed_right);
639 
640  //Speed regulator delay
641  DELAY_MS(25);
642 
643  } while (drive_cnt<encoder_cnt);
644 
645 
646  //Stop motors And jump back after litle delay
647  MOTOR_STOP();
648  DELAY_MS(500);
649 }
